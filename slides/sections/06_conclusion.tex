\section{Kết luận}

\begin{frame}{Tổng kết}
  \begin{itemize}
    \item Design Patterns là công cụ, không phải mục đích
    \item 3 nhóm: Creational, Structural, Behavioral
    \item Mỗi pattern giải quyết vấn đề cụ thể
    \item Áp dụng đúng lúc, đúng chỗ
    \item Cân nhắc trade-offs
  \end{itemize}
\end{frame}

\begin{frame}{Key Takeaways}
  \begin{enumerate}
    \item Hiểu vấn đề trước khi áp dụng pattern
    \item Patterns giúp giao tiếp hiệu quả
    \item Không over-engineer
    \item Kết hợp patterns khi phù hợp
    \item Thực hành để nắm vững
  \end{enumerate}
\end{frame}

\begin{frame}{Tài liệu tham khảo}
  \begin{itemize}
    \item \textit{Design Patterns: Elements of Reusable Object-Oriented Software} (Gang of Four)
    \item \textit{Head First Design Patterns} (Freeman \& Freeman)
    \item Refactoring.Guru
    \item SourceMaking.com
    \item GitHub repositories với implementations
  \end{itemize}
\end{frame}
